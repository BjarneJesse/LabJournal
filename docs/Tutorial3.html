1.

class12b_igraph <- graph_from_data_frame(
  d = class12b %>%
    select(namenr, age, sex, friend1:friend12) %>%
    pivot_longer(
      cols = friend1:friend12,
      names_to = "sourcevar",
      values_to = "to"
    ) %>%
    filter(!is.na(to)) %>%
    rename(from = namenr) %>%
    relocate(to, .after = from),
  
  vertices = class12b %>%
    select(namenr, age, sex),
  
  directed = TRUE
)
