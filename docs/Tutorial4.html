PupilsWaveV <- read_dta(file = "PupilsWaveV.dta")
class12b <- PupilsWaveV %>% 
  filter(schoolnr == "12b")  # keep only one class

class12b <- class12b %>%
  mutate(
    acthomew = as.numeric(zap_labels(acthomew)),
    acthomew = na_if(acthomew, 9),
    acthomew_rev = 6 - acthomew,
    sex = as.numeric(zap_labels(sex)),
    female = ifelse(sex == 1, 1, 0)
  )

class12b_igraph <- graph_from_data_frame(
  d = class12b %>%
    select(namenr, age, sex, friend1:friend12) %>%
    pivot_longer(
      cols = friend1:friend12,
      names_to = "sourcevar",
      values_to = "to"
    ) %>%
    filter(!is.na(to)) %>%
    rename(from = namenr) %>%
    relocate(to, .after = from),
  
  vertices = class12b %>%
    select(namenr, age, sex),
  
  directed = TRUE
)




class12b_adj_matrix <- as_adjacency_matrix(class12b_igraph, sparse = FALSE)
